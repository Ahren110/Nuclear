HTML+CSS+JS

HTML片段+局部CSS+JS逻辑代码组成可复用的组件。布局、样式、数据、逻辑独立，让程序易维护、可扩展

超小的体积

nuclear.min.js – 5KB (gzip)

双向绑定

使用observejs进行数据变更监听通知视图刷新，空间换取时间

局部CSS

提供style方法任性书写CSS，只针对自身组件生效，不污染其他组件

任意模板

默认采用了mustachejs（无逻辑模版），Nuclear提供了render方法组装拼接模板提高模板的可编程性。也可通过重写Nuclear.render使用其他模板引擎

松耦合

Nuclear是class base的框架，拥有OO的三大特性封装继承多态，更方便地抽象现实世界的物体及其属性或者逻辑算法

一睹为快

1：一个组件会依赖option，上面的option是{items:[]}；
2：代码里可以通过this.option访问其下面所有属性。option的变更会自动更新dom
3：模板里可以直接访问到option的属性
4：add方法不用担心会调用全局的，Nuclear会帮你处理好，add就是调用自身的add。
5：button和h3的样式都是局部的，不污染其他组件
6：可以通过标记nc-id的方式快速访问dom元素
其他 ：如果你讨厌反斜杠可以点击这里看看

反斜杠解药:使用template标签

<div id="todoListContainer"></div>

<template id="myTemplate">
    <style scoped>
        h3 { color:red; }
        button{ color:green;}
    </style>

    <div>
        <div>
            <h3>TODO</h3>
            <ul>{{#items}}<li>{{.}}</li>{{/items}}</ul>
            <form onsubmit="add(event)">
                <input nc-id="textBox" type="text">
                <button>Add #{{items.length}}</button>
            </form>
        </div>
    </div>
</template>

<script>
    var TodoApp = Nuclear.create({
        install:function() {
            this.todoTpl = document.querySelector("#myTemplate").innerHTML;
        },
        add: function (evt) {
            evt.preventDefault();
            this.option.items.push(this.textBox.value);
        },
        render: function () {
            return this.todoTpl;
        }
    });

    new TodoApp({ items: [] }, "#todoListContainer");
</script>
</script>

不用担心template标签的兼容性问题，Nuclear帮你处理好了。支持所有现代浏览器(包括IE9+)

自动局部刷新

var MarkdownEditor = Nuclear.create({
    install: function () {
        this.option.html = marked(this.option.value);
    },
    keyup: function () {
        this.option.value = this.textarea.value;
        this.option.html = marked(this.textarea.value);
    },
    render: function () {
        return '<div>\
            <h3>Input</h3>\
            <textarea nc-id="textarea" onkeyup="keyup()" rows="10" cols="25">{{value}}</textarea>\
            <h3>Output</h3>\
            <div class="content" >\
                {{{html}}}\
            </div>\
        </div>';

    }
});

new MarkdownEditor({ value: 'Type some *markdown* here!', html: '' }, "#markdownContainer");

你不用告诉Nuclear你要刷新什么地方，Nuclear会自动帮你dom diff并且替换掉变动的地方。Nuclear可以很轻松和任何框架/库搭配使用，如上面的markedjs。

组件继承

var Button = Nuclear.create({
    install: function () {
        this.option.disable = false;
    },
    click: function (evt) {
        if (this.option.click) {
            this.option.click.call(this, evt);
        }
    },
    render: function () {
        return ' <a onclick="click(event)" class="btn {{#disable}}disable{{/disable}}">{{text}}</a>';
    }
})

var SwitchButton = Button.create({
    click: function (evt) {
        this.option.disable = !this.option.disable;
        if (this.option.disable) {
            this.option.close && this.option.close.call(this, evt);
        } else {
            this.option.open && this.option.open.call(this, evt);
        }
    }
})


new Button({
    text: "普通按钮",
    click: function (evt) {
        console.log(this.option.text);//output: 普通按钮
    }
}, "#buttonContainer");

new SwitchButton({
    text: "开关按钮",
    open: function () { },
    close: function () { }
}, "#switchButtonContainer");

通过 Button.create方法可以创建继承 Button的组件。可以看到子类没有定义render方法，因为它们可以共用一套模板。当然，你也可以重新定义render，也可以通过 this._super访问到父类的所有方法。

var SVGApple = Nuclear.create({
    toggle: function () {
        if (this.playing) {
            window.cancelAnimationFrame(this.ticker);
            this.playing = false;
        } else {
            this.draw();
            this.playing = true;
        }
    },
    draw: function () {
        this.currentOffset += this.step;
        this.currentPath.style.strokeDashoffset = this.currentOffset;
        this.ticker = window.requestAnimationFrame(this.draw.bind(this));
        if (this.currentOffset < 0) {
            this.step *= -1;
            this.currentOffset = 0;
        }
    },
    installed: function () {
        this.length = this.currentPath.getTotalLength();
        this.currentPath.style.strokeDasharray = this.length + ' ' + this.length;
        this.currentPath.style.strokeDashoffset = this.length;
        this.currentOffset = this.length;
        this.ticker = 0;
        this.step = -1;
        this.playing = true;
        this.draw();
    },
    render: function () {
