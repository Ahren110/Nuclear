Nuclear.create

核心方法，用于创建组件

语法

var AnyComponent = Nuclear.create(obj, [setting]);

参数

参数 类型 描述 是否必须
object 对象字面量 组件的配置信息对象，一般包含render、style等方法 是
object 对象字面量 组件的设置，twoWay来设置是否双向绑定，默认为打开；diff用来设置是否dom diff自动局部刷新，默认为打开 否，可选参数

返回值

返回值 类型 描述
AnyComponent 类型Component类 组件类，可以通过new AnyComponent()去实例化该类

实例

var TodoApp = Nuclear.create({
    add: function (evt) {
        evt.preventDefault();
        this.option.items.push(this.textBox.value);
    },
    render: function () {
        return '<div>\
                    <h3>TODO</h3>\
                    <ul> {{#items}} <li>{{.}}</li> {{/items}}</ul>\
                    <form onsubmit="add(event)" >\
                        <input nc-id="textBox" type="text"  />\
                        <button>Add #{{items.length}}</button>\
                    </form>\
                </div>';
    }
});

new TodoApp({ items: [] }, "#todoListContainer");

Nuclear.render

重写该方法用于模板引擎替换

实例

Nuclear.render = function(tpl, data){
    return template.compile(tpl)(data);
};

var ArtTemplateApp = Nuclear.create({
    render: function () {
        return '<div>\
                    <h1>{{title}}</h1>\
                    <ul>\
                    {{each list as value i}}\
                        <li>索引 {{i + 1}} ：{{value}}</li>\
                    {{/each}}\
                    </ul>\
                </div>';
    }
});

new ArtTemplateApp( {
    title: '标签',
    list: ['文艺', '博客', '摄影', '电影', '民谣', '旅行', '吉他']
},"#container");

Nuclear.render在Nuclear内部调用，不需要使用者调用。使用者可以选择重写该方法替换模板引擎。该方法接收两个参数，tpl是模板，data是数据，返回html字符串。上面的例子是把默认的模板引擎替换成 artTemplate。

install

组件组装

实例

var HelloMessage = Nuclear.create({
    install: function () {
        this.option.name = this.option.firstName + this.option.lastName;
    },
    render: function () {
        return '<div>Hello ,My name is {{name}} !</div>';
    }
})

var hm=new HelloMessage( { firstName: "Zhang" ,lastName:"Lei" },"#helloContainer");

install方法会在nuclear内部优先执行，执行完才会进行属性监听。所以很明显，上面的name属性也进行了双向绑定。hm.option.name的修改会自动修改Dom。

style

组件样式

返回值

类型 描述
字符串 必须符合css格式和语法，用于渲染组件的样式

实例

var HelloMessage = Nuclear.create({
    style: function () {
        return '.test a{ \
                    color: red;\
                    font-size: 100px;\
                }\
                span{\
                        background-color: green;\
                }';
    },
    render: function () {
        return '<div>\
                    <span class="test" >\
                        <a href="#">Test Style</a>\
                    </span>\
                </div>';
    }
})

var hm = new HelloMessage( {  },"#helloContainer");

style方法会在nuclear内部执行，返回值必须是CSS格式的字符串。在style方法里面可以任性随意得写CSS选择器而不用担心污染其他组件。

render

组件模板结构

返回值

类型 描述
字符串 必须符合模板引擎语法，用于渲染组件的结构

如果没有重写Nuclear.render方法，则按照mustachejs的语法写就可以。

installed

组件组装完成执行

onRefresh

组件刷新执行的回调

onOptionChange

组件option变更执行的回调

参数

参数 类型 描述
prop 字符串 组件option被更改的属性名
value object 组件option更新的值
oldValue 任意对象 组件option更新前的值
path 字符串 组件option更改的path路径，用来描述option树中的拿个属性被更改

返回值

类型 描述
Boolean 返回true代表会重新刷新dom节点，返回false或者undefined代表不触发dom的刷新（这样的话，刷新dom的逻辑可以自己写在onOptionChange里）

实例

var Carousel = Nuclear.create({
    installed: function () {
        this.links = this.nav.querySelectorAll('a');
        this.active();
    },
    to: function (index) {
        this.option.index = index;
    },
    active: function () {
        util.removeClass(this.links, "active");
        util.addClass(this.links[this.option.index], "active");
    },
    onOptionChange: function (prop, value, oldValue, path) {
        if (prop === "index") {
            this.carouselScroll.style.left = value * -100 + "%";
            this.active();
            return false;
        }
    },
    style: function () {
        return ".nuclear-carousel-box { transition: left 1s cubic-bezier(0.22, 0.61, 0.36, 1); }";
    },
    render: function () {
        var imgCount = this.option.imgs.length;
        return '<div class="nuclear-carousel">\
                    <div style="width: ' + imgCount * 100 + '%; left: ' + this.option.index * -100 + '%;" nc-id="carouselScroll" class="nuclear-carousel-box">\
                        {{#imgs}}<img style=" width:'+ 100 / imgCount + '%;" src="{{.}}" /> {{/imgs}}\
                    </div>\
                    <div nc-id="nav"  class="nuclear-nav"> \
                        {{#imgs}}<a onclick="to({{@index}})"></a> {{/imgs}}</div>\
                </div>';
    }
});

new Carousel({
    imgs: ["img/room.jpg", "img/sleep.jpg", "img/watch.jpg"],
    index: 0
}, "#carouselContainer");

个人比较喜欢这个例子，这个例子用了大部分Nuclear特性。可以看到onOptionChange当检测到变更属性为index的时候返回false阻止dom自动刷新，自己手动更改left，利用style方法内定义好的transition进行运动。

option

组件的配置

类型

JSON/对象字面量

实例

var Selector = Nuclear.create({
    anyMethod: function () {
        console.log(this.option); //Object {item: Array[4], selectedIndex: 2}
    },
    render: function () {
        //省略
        return "";
    },
    style: function () {
        //省略
    }
});

var selector = new Selector({
    item: [
        { value: "all", text: "累计" },
        { value: "2015-9", text: "2015年9月" },
        { value: "2015-8", text: "2015年8月" },
        { value: "2015-7", text: "2015年7月" }
    ],
    selectedIndex: 2,
    change: function (item, index) {
    }
}, "#selectorContainer");

在任意方法内都可以通过this.option访问用户传入的组件配置。这里需要注意：install方法内可以改变this.option，install执行之后才会进行option的双向绑定。

node

组件的根节点

实例

var AnyThing = Nuclear.create({
    anyMethod: function () {
        /*输出
        <div data-nuclearid="0">
            <h3>我不是this.node</h3>
            <p>我也不是this.node</p>
        </div>
        */
        console.log(this.node); 
    },
    render: function () {
        //省略
        return '<h3>我不是this.node</h3>\
                <p>我也不是this.node</p>';
    }
});

不管render里的模板是不是放在一个封闭的div里。nuclear都会帮你会在最外层加上div包起来并可以通过this.node访问该根节点。

refresh

刷新组件

当遇到超大数据列表，开发者关闭twoWay双向绑定的时候，可以通过该方法手动刷新组件。如这个性能测试的例子就关闭了双向绑定进行手动刷新。

nc-id

通过this可以访问到该节点
var MarkdownEditor = Nuclear.create({
    install: function () {
        this.option.html = marked(this.option.value);
    },
    keyup: function () {
        this.option.html = marked(this.textarea.value);
    },
    render: function () {
        return '<div>\
                    <h3>Input</h3>\
                    <textarea nc-id="textarea" onkeyup="keyup()" rows="10" cols="25">{{value}}</textarea>\
                    <h3>Output</h3>\
                    <div class="content" nc-refresh >\
                        {{{html}}}\
                    </div>\
                </div>';
    }
});

如上面所示，通过this.textarea可以访问到标记nc-id为textarea的节点。

nc-class

通过this可以访问到存储所有标记该类class节点的数组

类型

Array
var AnyThing = Nuclear.create({
    anyMethod: function () {
        console.log(this.abc); //输出 [h3 , p]
    },
    render: function () {
        //省略
        return '<h3 nc-class="abc">我不是this.node</h3>\
                <p nc-class="abc">我也不是this.node</p>';
    }
});

如上面所示，通过this.abc可以访问到存有所有标记nc-class为abc的数组。
